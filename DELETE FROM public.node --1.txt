DELETE FROM public.node --1
	WHERE  is_deleted= true 


	DELETE FROM public.line --2
USING public.node                     
WHERE line.id = node."id" 
  AND node.node_type = 'elevator';
  
delete from public.line using  public.floor--3
Where line.id = floor."id" 
and  floor.is_deleted is false

DELETE FROM public.floor --4
USING public.venue
WHERE floor.id = venue.id
  AND floor.level = 0
  AND venue.id = 1;



DELETE FROM public.venue v --5
WHERE NOT EXISTS (
  SELECT 1
  FROM public.floor f
  WHERE f.venue_id = v.id
);




  
  DELETE FROM public.line-6
USING public.node
WHERE 
  (line.first_node_id = node.id OR line.second_node_id = node.id)
  AND node.node_type = 'stairs';
